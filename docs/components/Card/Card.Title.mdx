## Usage

```tsx
import { Card } from "@/components";
import React from "react";

const Example = () => {
  return (
    <Card>
        <Card.Title title="This is card title" />
    </Card>
  );
};

export default Example;
```

## Props

Extends the [`TextProps<Theme>`](/docs/components/Text) from text compnent

### title
**Type:** `string`
