## Usage

```tsx
import { Card } from "@/components";
import React from "react";

const Example = () => {
  return (
    <Card>
      <Card.CardCover />
    </Card>
  );
};

export default Example;
```

## Props

Extends the [`ImageBannerProps`](/docs/components/ImageBanner) from the ImageBanner Component
